3//! This module provides the `RenderContext` type, which encapsulates
4//! rendering-related computations and state for text sprites.
5//!
6//! # Purpose
7//! The `RenderContext` simplifies and centralizes text rendering logic by
8//! managing:
9//! - Font assets, texture atlases, and associated metadata.
10//! - Cached computations for glyph dimensions, text width, and alignment
11//!   offsets.
12//! - Handles text positioning and scaling for both `atlas_sprites`
13//!   (sprite-based) and `rendered` (pre-rendered image-based) text rendering.
14//!   This centralizes font asset access, text dimension calculations, and
15//!   alignment logic.
16//!
17//! By consolidating these responsibilities, `RenderContext` reduces code
18//! duplication and makes rendering logic more maintainable.
19//!
20//! # Key Features
21//! - **Asset Management**: Fetches and stores references to the font and
22//!   texture atlas assets.
23//! - **Text Calculations**: Computes text dimensions, glyph dimensions, and
24//!   scaling factors.
25//! - **Unified Rendering Support**: Handles both `atlas_sprites` (sprite-based)
26//!   and `rendered` (pre-rendered image-based) text rendering.
27//! - **Caching**: Uses `CacheCell` to lazily compute and store values like
28//!   maximum height and anchor offsets.
29//!
30//! This module is intended for internal use within the text rendering system
31//! and is designed to work seamlessly with other components, such as
32//! `SpriteContext`.

48/// Groups font-related assets and configuration for rendering text sprites.
49///
50/// Includes references to the texture atlas layout, font asset, and the
51/// font text component that defines the text content and font height.
52pub(crate) struct RenderContext<'assets> {
53    /// The texture atlas layout defining glyph placements.
54    atlas_layouts: Vec<&'assets TextureAtlasLayout>,
55    /// The font asset containing glyph metadata.
56    image_font: &'assets ImageFont,
57    /// The text component defining the content and font height.
58    image_font_text: &'assets ImageFontText,
59    /// Configuration for rendering the text, including anchor alignment,
60    /// color, letter spacing, and scaling behavior.
61    pub render_config: RenderConfig,
62    /// The text filtered to include only supported characters in the font
63    /// atlas.
64    filtered_text: FilteredString<'assets, &'assets String>,
65
66    /// Cached maximum glyph height.
67    max_height: CacheCell<u32>,
68}

70impl<'assets> RenderContext<'assets> {
71    /// Creates a new `RenderContext` for rendering text using an `ImageFont`.
72    ///
73    /// This function retrieves the necessary assets, filters the text to
74    /// exclude unsupported characters, and initializes cached computations
75    /// for rendering.
76    ///
77    /// # Parameters
78    /// - `image_font`: A reference to the loaded `ImageFont` asset.
79    /// - `image_font_text`: A reference to the `ImageFontText` component
80    ///   containing the text.
81    /// - `render_config`: Rendering options such as color, anchor alignment,
82    ///   letter spacing, and scaling mode.
83    /// - `texture_atlas_layouts`: The asset collection containing texture atlas
84    ///   layouts.
85    ///
86    /// # Returns
87    /// - `Some(RenderContext)`: If all required assets are available.
88    /// - `None`: If the font or texture atlas layouts are missing.
89    pub(crate) fn new(
90        image_font: &'assets ImageFont,
91        image_font_text: &'assets ImageFontText,
92        render_config: RenderConfig,
93        texture_atlas_layouts: &'assets Assets<TextureAtlasLayout>,
94    ) -> Option<Self> {
95        let atlas_layouts: Result<Vec<_>, _> = image_font
96            .atlas_layouts
97            .iter()
98            .map(|texture_atlas_layout| {
99                texture_atlas_layouts
100                    .get(texture_atlas_layout)
101                    .ok_or_else(|| {
102                        format!("TextureAtlasLayout not loaded: {texture_atlas_layout:?}")
103                    })
104            })
105            .collect();
106
107        let atlas_layouts = match atlas_layouts {
108            Ok(layout) => layout,
109            Err(error) => {
110                error!("{error}");
111                return None;
112            }
113        };
114
115        let filtered_text = image_font.filter_string(&image_font_text.text);
116
117        Some(RenderContext {
118            atlas_layouts,
119            image_font,
120            image_font_text,
121            render_config,
122            filtered_text,
123
124            max_height: default(),
125        })
126    }

128    /// Computes the uniform scaling factor for text glyphs.
129    ///
130    /// Determines the scaling factor to apply to glyph dimensions based on
131    /// the specified font height and the maximum glyph height.
132    ///
133    /// # Returns
134    /// An `f32` representing the uniform scaling factor for text sprites.
135    #[expect(
136        clippy::cast_precision_loss,
137        reason = "`max_height` won't ever be particularly large"
138    )]
139    #[inline]
140    pub(crate) fn scale(&self) -> f32 {
141        let max_height = self.max_height();
142        self.image_font_text
143            .font_height
144            .map_or(1.0, |font_height| font_height / max_height as f32)
145    }
146
147    /// Calculates the maximum height of the filtered text.
148    ///
149    /// Iterates over the filtered text characters to determine the overall
150    /// height based on glyph sizes in the texture atlas.
151    ///
152    /// # Returns
153    /// The height of the tallest glyph
154    #[inline]
155    pub(crate) fn max_height(&self) -> u32 {
156        self.max_height.get_or_insert_with(|| {
157            let mut max_height = 1;
158
159            for character in self.filtered_text.filtered_chars() {
160                let image_font_character = &self.image_font.atlas_character_map[&character];
161                let rect = self.atlas_layouts[image_font_character.page_index].textures
162                    [image_font_character.character_index];
163                max_height = max_height.max(rect.height());
164            }
165
166            max_height
167        })
168    }
169
170    /// Calculates the total width of the rendered text based on the filtered
171    /// characters and glyph dimensions stored in the context.
172    ///
173    /// # Returns
174    /// The total width of the rendered text, in pixels, after applying the
175    /// scaling factor.
176    #[inline]
177    pub(crate) fn text_width(&self) -> f32 {
178        let mut text_width = 0.;
179
180        for character in self.filtered_text.filtered_chars() {
181            let (width, _) = self.character_dimensions(character);
182            text_width += width;
183        }
184        text_width
185    }

187    /// Computes the dimensions of a glyph for a given character, applying
188    /// scaling if a specific font height is provided.
189    ///
190    /// The dimensions are determined using the character's bounding rectangle
191    /// in the texture atlas, the configured letter spacing, and the
192    /// selected `ScalingMode`. Additionally, if
193    /// `RenderConfig::apply_scaling` is `true`, width scaling is
194    /// applied before rounding or truncation, ensuring consistent proportions
195    /// in certain scaling modes.
196    ///
197    /// # Parameters
198    /// - `character`: The character whose dimensions are being computed.
199    ///
200    /// # Returns
201    /// A tuple `(width, height)` representing the computed dimensions of the
202    /// glyph, where width scaling behavior depends on
203    /// `RenderConfig::apply_scaling`.
204    #[expect(
205        clippy::cast_precision_loss,
206        reason = "the magnitude of the numbers we're working on here are too small to lose anything"
207    )]
208    pub(crate) fn character_dimensions(&self, character: char) -> (f32, f32) {
209        let image_font_character = &self.image_font.atlas_character_map[&character];
210        let rect = self.atlas_layouts[image_font_character.page_index].textures
211            [image_font_character.character_index];
212        let letter_spacing = self.render_config.letter_spacing;
213        let width = rect.width() as f32 + letter_spacing;
214        let height = rect.height() as f32;
215        let max_height = self.max_height() as f32;
216
217        if let Some(font_height) = self.image_font_text.font_height {
218            if self.render_config.apply_scaling {
219                let scaling_mode = self.render_config.scaling_mode;
220                let scale_factor = font_height / max_height;
221                return (
222                    scaling_mode.apply_scale(width, scale_factor),
223                    scaling_mode.apply_scale(height, scale_factor),
224                );
225            }
226        }
227
228        (width, height)
229    }

348            .to_anchor_offsets(self.render_config.offset_characters)

351    /// Computes the transform for positioning and scaling a text sprite.
352    ///
353    /// This method calculates the sprite's translation and scale based on:
354    /// - The x-position of the sprite.
355    /// - The dimensions of the character's glyph.
356    /// - The alignment offsets and scaling configuration.
357    ///
358    /// # Parameters
359    /// - `x_pos`: A mutable reference to the current x-position of the sprite.
360    ///   This value is updated to reflect the position of the next sprite.
361    /// - `character`: The character associated with the sprite.
362    ///
363    /// # Returns
364    /// A [`Transform`] representing the position and scale of the sprite.
365    #[inline]
366    pub(crate) fn transform(&self, x_pos: &mut f32, character: char) -> Transform {
367        let x = *x_pos;
368        let (width, height) = self.character_dimensions(character);
369        *x_pos += self.character_x_advance(character).unwrap_or(width);
370
371        let params = ComputeTransformParams {
372            x_pos: x,
373            scaled_text_width: self.text_width(),
374            scaled_width: width,
375            scaled_height: height,
376            max_height: self.max_height(),
377            character_offsets: self.character_offsets(character),
378            scale: self.scale(),
379        };
380        self.anchor_offsets().compute_transform(params)
381    }

416    /// This value is specified at the fontâ€™s native height and is scaled

435/// A lightweight wrapper around a [`Cell<Option<T>>`] for caching values.
436///
437/// This utility type provides a mechanism to lazily compute and cache a value
438/// using a factory function. It is ideal for scenarios where a value is
439/// expensive to compute but may not always be needed.
440///
441/// # Features
442/// - Lazily initializes the cached value when accessed for the first time.
443/// - Ensures the cached value is reused across multiple accesses.
444/// - Supports types that implement the [`Copy`] trait.
445struct CacheCell<T>(Cell<Option<T>>);
446
447impl<T> CacheCell<T> {
448    /// Retrieves the cached value, computing it if necessary.
449    ///
450    /// If the value is already cached, this method returns it directly. If not,
451    /// it invokes the provided factory function to compute the value, caches
452    /// it, and then returns it.
453    ///
454    /// # Parameters
455    /// - `factory`: A closure that computes the value if it is not already
456    ///   cached.
457    ///
458    /// # Returns
459    /// The cached value, or the newly computed value if the cache was empty.
460    fn get_or_insert_with<F>(&self, factory: F) -> T
461    where
462        T: Copy,
463        F: FnOnce() -> T,
464    {
465        if let Some(value) = self.0.get() {
466            value
467        } else {
468            let value = factory();
469            self.0.set(Some(value));
470            value
471        }
472    }
473}
474
475impl<T> Default for CacheCell<T> {
476    /// Creates a new, empty `CacheCell`.
477    ///
478    /// The cache will initialize with no value and will compute the value on
479    /// first access via `get_or_insert_with`.
480    fn default() -> Self {
481        Self(default())
482    }
483}
