5//! A module for filtering strings based on a character map.
6//!
7//! This module provides the [`FilteredString`] type, which offers a reusable
8//! and efficient abstraction for working with strings filtered according to a
9//! predefined character map. This is particularly useful in scenarios where
10//! only a subset of characters are supported, such as rendering text with an
11//! `ImageFont`.
12//!
13//! # Key Features
14//! - **Efficient Filtering:** Filters strings without unnecessary allocations,
15//!   making it suitable for performance-sensitive contexts.
16//! - **Iterator Support:** Allows iteration over filtered characters for
17//!   further processing.
18//! - **Display Implementation:** Can be directly converted to a string
19//!   representation of the filtered content.
20//!
21//! # Usage
22//! This module is typically used in conjunction with image font rendering
23//! systems, where only characters supported by the font's atlas are allowed. It
24//! ensures unsupported characters are ignored while preserving the order of the
25//! valid characters.

33/// A wrapper type for filtering characters from a string based on a character
34/// map.
35///
36/// This type allows you to filter out characters from a string that do not
37/// exist in the given `atlas_character_map`. It provides a reusable abstraction
38/// for working with filtered strings efficiently, avoiding unnecessary
39/// allocations.
40#[derive(Debug)]
41pub(crate) struct FilteredString<'map, S: AsRef<str>> {
42    /// The input string to be filtered.
43    ///
44    /// This string serves as the source for filtering operations. Characters
45    /// in this string are compared against the characters in
46    /// `atlas_character_map`.
47    string: S,
48
49    /// A reference to a map of characters to their indices in a texture atlas.
50    ///
51    /// This map determines which characters from `string` are retained during
52    /// filtering. Only characters present as keys in this map will be included
53    /// in the filtered output.
54    atlas_character_map: &'map HashMap<char, ImageFontCharacter>,
55}
56
57impl<'map, S: AsRef<str>> FilteredString<'map, S> {
58    /// Creates a new `FilteredString` instance.
59    ///
60    /// # Parameters
61    /// - `string`: The input string to be filtered.
62    /// - `atlas_character_map`: A reference to a character map that determines
63    ///   which characters are retained.
64    ///
65    /// # Returns
66    /// A `FilteredString` instance that can produce iterators over the filtered
67    /// characters.
68    pub(crate) fn new(
69        string: S,
70        atlas_character_map: &'map HashMap<char, ImageFontCharacter>,
71    ) -> Self {
72        Self {
73            string,
74            atlas_character_map,
75        }
76    }
77
78    /// Returns an iterator over the filtered characters.
79    ///
80    /// This method filters the input string to include only characters that
81    /// exist in the `atlas_character_map`.
82    ///
83    /// # Returns
84    /// An iterator that yields characters retained by the filter.
85    pub(crate) fn filtered_chars(&self) -> impl Iterator<Item = char> + '_ {
86        self.string
87            .as_ref()
88            .chars()
89            .filter(|character| self.atlas_character_map.contains_key(character))
90    }
91
92    /// Checks if the filtered string is empty.
93    ///
94    /// # Returns
95    /// `true` if there are no characters in the filtered string; otherwise,
96    /// `false`.
97    #[inline]
98    pub(crate) fn is_empty(&self) -> bool {
99        self.filtered_chars().next().is_none()
100    }
101}
102
103impl<S: AsRef<str>> fmt::Display for FilteredString<'_, S> {
104    fn fmt(&self, formatter: &mut fmt::Formatter<'_>) -> fmt::Result {
105        for character in self.filtered_chars() {
106            write!(formatter, "{character}")?;
107        }
108        Ok(())
109    }
110}

112impl ImageFont {
113    /// Filters a string to include only characters present in the font's
114    /// character map.
115    ///
116    /// This function returns a
117    /// [`FilteredString`](filtered_string::FilteredString) containing only the
118    /// characters from the input string that exist in the font's
119    /// `atlas_character_map`. It ensures that unsupported characters are
120    /// excluded during rendering.
121    ///
122    /// # Parameters
123    /// - `string`: The input string to filter.
124    ///
125    /// # Returns
126    /// A `FilteredString` returning only characters supported by the font.
127    ///
128    /// # Notes
129    /// This function requires either the `rendered` or `atlas_sprites` feature
130    /// to be enabled.
131    pub(super) fn filter_string<S: AsRef<str>>(&self, string: S) -> FilteredString<'_, S> {
132        FilteredString::new(string, &self.atlas_character_map)
133    }
134}
