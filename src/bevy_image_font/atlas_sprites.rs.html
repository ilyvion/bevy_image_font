1//! This module provides functionality for rendering text as individual sprites
2//! using the Bevy engine, utilizing custom image fonts.
3//!
4//! It breaks down text into individual characters and represents them as
5//! sprites in the game world. This approach allows precise positioning and
6//! styling of text at the character level, suitable for scenarios where text
7//! needs to be rendered dynamically or interactively.
8//!
9//! Key Features:
10//! - `ImageFontSpriteText` component: Allows customization of text rendering,
11//!   such as color and anchor point.
12//! - Systems for rendering text to sprite entities and updating their
13//!   configuration when text changes.
14//! - Optional gizmo rendering for debugging purposes, available with the
15//!   "gizmos" feature flag.
16//!
17//! This module is intended for advanced text rendering use cases, offering
18//! fine-grained control over how text is displayed in the game world.
19

56/// Text rendered using an [`ImageFont`] as individual sprites.
57///
58/// This struct provides fields for customizing text rendering, such as
59/// alignment, color, and scaling behavior.
60///
61/// - `anchor`: Specifies the alignment point of the text relative to its
62///   position.
63/// - `color`: Uniform tint applied to all glyphs.
64/// - `scaling_mode`: Controls how scaling is applied to glyph dimensions.
65#[derive(Debug, Clone, Reflect, Default, Component, Setters)]
66#[setters(into)]
67#[require(ImageFontText, Visibility)]
68#[non_exhaustive]
69pub struct ImageFontSpriteText {
70    /// The alignment point of the text relative to its position. For example,
71    /// `Anchor::TopLeft` aligns the text's top-left corner to its position.
72    pub anchor: Anchor,
73
74    /// The color applied to the rendered text. This color affects all glyphs
75    /// equally, allowing you to tint the text uniformly.
76    pub color: Color,
77
78    /// Determines how scaling is applied to the glyph dimensions when adjusting
79    /// them to match the desired font height.
80    ///
81    /// This field allows control over how fractional scaling values are
82    /// handled, using the [`ImageFontScalingMode`] enum. It provides options to
83    /// truncate, round, or retain precise fractional values, depending on
84    /// your rendering requirements.
85    ///
86    /// The default value is `ImageFontScalingMode::Rounded`.
87    pub scaling_mode: ImageFontScalingMode,
88
89    /// Determines a constant kerning between characters. The spacing is given
90    /// at the font's native height and is scaled proportionally based on the
91    /// current font height.
92    pub letter_spacing: LetterSpacing,
93}

95/// Stores a mapping between characters and their corresponding sprite entities.
96/// This is used to manage text rendering at the entity level.
97#[derive(Debug, Clone, Component)]
98struct ImageFontTextData {
99    /// The entity that owns this `ImageFontTextData` component.
100    self_entity: Entity,
101
102    /// Basically a map between character index and character sprite
103    sprites: Vec<Entity>,
104
105    /// Tracks whether a missing font asset has already been reported for this
106    /// entity.
107    ///
108    /// This flag prevents repeated error messages when an `ImageFontText`
109    /// component references a font asset that has not been loaded. Once an
110    /// error is logged for a missing font, subsequent frames will not log
111    /// additional errors for the same entity.
112    ///
113    /// # Default Behavior
114    /// - Initially set to `false`, meaning no missing font error has been
115    ///   reported.
116    /// - Set to `true` after the first error message is logged.
117    has_reported_missing_font: bool,
118
119    /// Stores debugging information for visualizing `ImageFontSpriteText` in
120    /// the scene.
121    ///
122    /// This field is only available when the `gizmos` feature is enabled. It
123    /// tracks glyph dimensions and other relevant data used for rendering
124    /// debug gizmos, such as bounding boxes and anchor points.
125    #[cfg(feature = "gizmos")]
126    gizmo_data: gizmos::ImageFontTextGizmoData,
127}
128impl ImageFontTextData {
129    /// Creates a new `ImageFontTextData` instance for a given entity.
130    ///
131    /// This function initializes an empty `ImageFontTextData` struct and
132    /// associates it with the provided entity. It ensures that newly
133    /// created text data always has a reference to its owning entity.
134    ///
135    /// # Parameters
136    /// - `self_entity`: The entity that this `ImageFontTextData` belongs to.
137    ///
138    /// # Returns
139    /// A new instance of `ImageFontTextData` with an empty sprite list and the
140    /// specified `self_entity`.
141    fn new(entity: Entity) -> Self {
142        Self {
143            self_entity: entity,
144            sprites: default(),
145            has_reported_missing_font: default(),
146            #[cfg(feature = "gizmos")]
147            gizmo_data: default(),
148        }
149    }
150}

152/// System that renders each [`ImageFontText`] as child [`Sprite`] entities,
153/// where each sprite represents a character in the text. Each sprite is
154/// positioned based on its order in the text, accounting for letter spacing,
155/// scaling mode, and anchor alignment. This system only runs when the
156/// `ImageFontText` or [`ImageFontSpriteText`] changes.
157#[expect(
158    clippy::missing_panics_doc,
159    reason = "expect() is only used on a newly created Some() value"
160)]
161#[expect(
162    private_interfaces,
163    reason = "Systems are only `pub` for the sake of allowing dependent crates to use them for ordering"
164)]
165pub fn set_up_sprites(
166    mut commands: Commands,
167    mut query: Query<
168        (
169            Entity,
170            &ImageFontText,
171            &ImageFontSpriteText,
172            Option<&mut ImageFontTextData>,
173        ),
174        Or<(Changed<ImageFontText>, Changed<ImageFontSpriteText>)>,
175    >,
176    mut child_query: Query<(&mut Sprite, &mut Transform)>,
177    image_fonts: Res<Assets<ImageFont>>,
178    texture_atlas_layouts: Res<Assets<TextureAtlasLayout>>,
179) {
180    for (entity, image_font_text, image_font_sprite_text, mut image_font_text_data) in &mut query {
181        let mut maybe_new_image_font_text_data = None;
182        let image_font_text_data = if let Some(image_font_text_data) = image_font_text_data.as_mut()
183        {
184            &mut *image_font_text_data
185        } else {
186            maybe_new_image_font_text_data = Some(ImageFontTextData::new(entity));
187            #[expect(clippy::expect_used, reason = "newly created Some() value")]
188            maybe_new_image_font_text_data
189                .as_mut()
190                .expect("newly created Some() value")
191        };
192
193        let render_config = RenderConfig {
194            letter_spacing: image_font_sprite_text.letter_spacing.to_f32(),
195            offset_characters: true,
196            apply_scaling: true,
197            scaling_mode: image_font_sprite_text.scaling_mode,
198            color: image_font_sprite_text.color,
199            text_anchor: image_font_sprite_text.anchor,
200        };
201
202        let font_handle = &image_font_text.font;
203        let Some(image_font) = image_fonts.get(font_handle) else {
204            if !image_font_text_data.has_reported_missing_font {
205                let font_handle_detail: &dyn Debug = if let Some(font_path) = font_handle.path() {
206                    font_path
207                } else {
208                    &font_handle.id()
209                };
210                error!(
211                    "ImageFont asset {font_handle_detail:?} is not loaded; can't render text for entity: {}",
212                    image_font_text_data.self_entity
213                );
214                image_font_text_data.has_reported_missing_font = true;
215            }
216            continue;
217        };
218
219        let Some(render_context) = RenderContext::new(
220            image_font,
221            image_font_text,
222            render_config,
223            &texture_atlas_layouts,
224        ) else {
225            maybe_insert_new_image_font_text_data(
226                &mut commands,
227                entity,
228                maybe_new_image_font_text_data,
229            );
230            continue;
231        };
232
233        let mut sprite_context = SpriteContext {
234            entity,
235            image_font_text_data,
236        };
237
238        let x_pos = update_existing_sprites(&mut child_query, &mut sprite_context, &render_context);
239
240        adjust_sprite_count(
241            x_pos,
242            &mut commands,
243            &mut sprite_context,
244            &render_context,
245            image_font_sprite_text,
246        );
247
248        maybe_insert_new_image_font_text_data(
249            &mut commands,
250            entity,
251            maybe_new_image_font_text_data,
252        );
253    }
254}

256/// Inserts a newly created `ImageFontTextData` component into an entity if one
257/// was generated.
258///
259/// This function takes an `Option<ImageFontTextData>`, which may contain a
260/// newly created instance. If it is `Some`, the function inserts the
261/// `ImageFontTextData` into the provided entity. If it is `None`, no action is
262/// taken.
263///
264/// # Parameters
265/// - `commands`: A command buffer used to insert the component if
266///   `maybe_new_image_font_text_data` is `Some`.
267/// - `entity`: The entity that will receive the `ImageFontTextData`.
268/// - `maybe_new_image_font_text_data`: An `Option<ImageFontTextData>` that
269///   contains a newly created instance if one was generated earlier.
270///
271/// # Notes
272/// - This function does not check whether the entity already has an
273///   `ImageFontTextData` component; it simply inserts the given value if it
274///   exists.
275/// - If a new component is inserted, a debug message is logged.
276fn maybe_insert_new_image_font_text_data(
277    commands: &mut Commands,
278    entity: Entity,
279    maybe_new_image_font_text_data: Option<ImageFontTextData>,
280) {
281    if let Some(new_image_font_text_data) = maybe_new_image_font_text_data {
282        debug!("Inserted new ImageFontTextData for entity {:?}", entity);
283        commands.entity(entity).insert(new_image_font_text_data);
284    }
285}

287/// Updates existing sprites to match the filtered text content.
288///
289/// Adjusts the position, scale, and appearance of each sprite to reflect
290/// the corresponding glyph in the text and texture atlas.
291///
292/// # Parameters
293/// - `child_query`: Query for accessing child sprite components.
294/// - `sprite_context`: Context for managing the entity and its sprite data.
295/// - `render_context`: Context providing rendering-related information and
296///   operations.
297///
298/// # Returns
299/// The x-position to the right of the last processed sprite.
300fn update_existing_sprites(
301    child_query: &mut Query<(&mut Sprite, &mut Transform)>,
302    sprite_context: &mut SpriteContext,
303    render_context: &RenderContext,
304) -> f32 {
305    let SpriteContext {
306        ref mut image_font_text_data,
307        ..
308    } = *sprite_context;
309
310    let mut x_pos = 0.;
311
312    for (sprite_entity, character) in image_font_text_data
313        .sprites
314        .iter()
315        .copied()
316        .zip(render_context.text().filtered_chars())
317    {
318        let (mut sprite, mut transform) = match child_query.get_mut(sprite_entity) {
319            Ok(result) => result,
320            Err(error) => {
321                error!(
322                    "An ImageFontSpriteText unexpectedly failed: {error}. This will likely cause rendering bugs."
323                );
324                continue;
325            }
326        };
327
328        let sprite = &mut *sprite;
329        let Some(sprite_texture) = sprite.texture_atlas.as_mut() else {
330            error!(
331                "An ImageFontSpriteText's child sprite was \
332            unexpectedly missing a `texture_atlas`. This will likely cause rendering bugs."
333            );
334            continue;
335        };
336
337        render_context.update_render_values(character, sprite_texture, &mut sprite.color);
338
339        *transform = render_context.transform(&mut x_pos, character);
340
341        #[cfg(feature = "gizmos")]
342        gizmos::record_character_dimensions(
343            render_context,
344            character,
345            &mut image_font_text_data.gizmo_data,
346        );
347    }
348
349    x_pos
350}

352/// Ensures the number of sprites matches the number of characters in the text.
353///
354/// Adds missing sprites or removes excess sprites to maintain consistency
355/// between the text content and the entity's children.
356///
357/// # Parameters
358/// - `x_pos`: x-position of where the next sprite should go.
359/// - `commands`: A command buffer for spawning or despawning sprites to
360///   synchronize with the text content.
361/// - `sprite_context`: Context for managing the entity and its sprite data.
362/// - `render_context`: Context providing rendering-related information and
363///   operations.
364/// - `sprite_text`: Component defining text appearance (e.g., color).
365#[inline]
366fn adjust_sprite_count(
367    x_pos: f32,
368    commands: &mut Commands,
369    sprite_context: &mut SpriteContext,
370    render_context: &RenderContext,
371    sprite_text: &ImageFontSpriteText,
372) {
373    use std::cmp::Ordering;
374
375    let char_count = render_context.text().filtered_chars().count();
376    let sprite_count = sprite_context.image_font_text_data.sprites.len();
377
378    match sprite_count.cmp(&char_count) {
379        Ordering::Greater => {
380            remove_excess_sprites(commands, sprite_context, char_count);
381        }
382        Ordering::Less => {
383            add_missing_sprites(x_pos, commands, sprite_context, render_context, sprite_text);
384        }
385        Ordering::Equal => {}
386    }
387}

389/// Removes excess sprites from the text entity to match the new character
390/// count.
391///
392/// # Parameters
393/// - `commands`: Command buffer for despawning entities.
394/// - `sprite_context`: Context for managing the entity and its sprite data.
395/// - `char_count`: The number of characters in the filtered text.
396///
397/// # Side Effects
398/// Excess sprites are despawned from the ECS.
399#[inline]
400fn remove_excess_sprites(
401    commands: &mut Commands,
402    sprite_context: &mut SpriteContext,
403    char_count: usize,
404) {
405    for entity in sprite_context
406        .image_font_text_data
407        .sprites
408        .drain(char_count..)
409    {
410        commands.entity(entity).despawn();
411    }
412}

414/// Adds missing sprites to the text entity to match the new character count.
415///
416/// If the number of sprites is less than the number of characters in the text,
417/// this function spawns new sprites for the remaining characters and updates
418/// the sprite data accordingly.
419///
420/// # Parameters
421/// - `x_pos`: x-position of where the next sprite should go.
422/// - `commands`: Command buffer for spawning new sprite entities.
423/// - `sprite_context`: Context for managing the entity and its sprite data.
424/// - `render_context`: Context providing rendering-related information and
425///   operations.
426/// - `sprite_text`: Component defining text appearance (e.g., color).
427fn add_missing_sprites(
428    mut x_pos: f32,
429    commands: &mut Commands,
430    sprite_context: &mut SpriteContext,
431    render_context: &RenderContext,
432    sprite_text: &ImageFontSpriteText,
433) {
434    let SpriteContext {
435        entity,
436        ref mut image_font_text_data,
437    } = *sprite_context;
438
439    let current_sprite_count = image_font_text_data.sprites.len();
440
441    commands.entity(entity).with_children(|parent| {
442        for character in render_context
443            .text()
444            .filtered_chars()
445            .skip(current_sprite_count)
446        {
447            let transform = render_context.transform(&mut x_pos, character);
448            let sprite = Sprite {
449                image: render_context.font_image(character),
450                texture_atlas: Some(render_context.font_texture_atlas(character)),
451                color: sprite_text.color,
452                ..Default::default()
453            };
454
455            let child = parent.spawn((sprite, transform));
456            image_font_text_data.sprites.push(child.id());
457
458            #[cfg(feature = "gizmos")]
459            gizmos::record_character_dimensions(
460                render_context,
461                character,
462                &mut image_font_text_data.gizmo_data,
463            );
464        }
465    });
466}

468/// Represents the entity and its associated text sprites during rendering.

470/// Manages the commands for modifying the entity, its sprite data, and the

474    entity: Entity,
475    /// The mutable text sprite data component for the entity.
476    image_font_text_data: &'data mut ImageFontTextData,
