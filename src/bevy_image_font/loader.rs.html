23/// Human-readable way to specify where the characters in an image font are.
24#[derive(Debug, Serialize, Deserialize)]
25#[non_exhaustive]
26pub enum ImageFontLayout {
27    /// Interprets the string as a "grid" and slices up the input image
28    /// accordingly. Leading and trailing newlines are stripped, but spaces
29    /// are not (since your font might use them as padding).
30    ///
31    /// ```rust
32    /// # use bevy_image_font::loader::*;
33    /// // note that we have a raw string *inside* a raw string here...
34    /// let s = r###"
35    ///
36    /// // this bit is the actual RON syntax
37    /// Automatic(r##"
38    ///  !"#$%&'()*+,-./0123
39    /// 456789:;<=>?@ABCDEFG
40    /// HIJKLMNOPQRSTUVWXYZ[
41    /// \]^_`abcdefghijklmno
42    /// pqrstuvwxyz{|}~
43    /// "##)
44    ///
45    /// "###;
46    /// let layout = ron::from_str::<ImageFontLayout>(s).unwrap();
47    /// ```
48    Automatic(String),
49
50    /// Manually specifies the top-left position of each character, where each
51    /// character has the same size. When writing this in RON, the syntax
52    /// will look like
53    ///
54    /// ```rust
55    /// # use bevy_image_font::loader::*;
56    /// let s = r#"
57    /// ManualMonospace(
58    ///   size: (4, 8),
59    ///   coords: {
60    ///      'a': (0, 0),
61    ///      'b': (10, 0)
62    ///   }
63    /// )
64    /// "#;
65    /// ron::from_str::<ImageFontLayout>(s).unwrap();
66    /// ```
67    ManualMonospace {
68        /// The size of each character, specified as a uniform width and height
69        /// in pixels. All characters are assumed to have the same dimensions.
70        size: UVec2,
71
72        /// A mapping from characters to their top-left positions within the
73        /// font image. Each position is given in pixel coordinates relative
74        /// to the top-left corner of the image.
75        coords: HashMap<char, UVec2>,
76    },
77
78    /// Fully specifies the bounds of each character. The most general case.
79    ///
80    /// ```rust
81    /// # use bevy_image_font::loader::*;
82    /// let s = r#"
83    /// Manual({
84    /// 'a': URect(min: (0, 0), max: (10, 20)),
85    /// 'b': URect(min: (20, 20), max: (25, 25))
86    /// })
87    /// "#;
88    /// ron::from_str::<ImageFontLayout>(s).unwrap();
89    /// ```
90    Manual(HashMap<char, URect>),
91}

93/// Errors that can show up during validation.
94#[derive(Debug, Error)]
95#[non_exhaustive]
96pub enum ImageFontLayoutValidationError {
97    /// The image width does not evenly divide the character count per line.
98    ///
99    /// This error occurs when the width of the provided image is not a multiple
100    /// of the number of characters per line specified in the layout.
101    #[error(
102        "Image width {width} is not an exact multiple of per-line character count \
103    {per_line_character_count}."
104    )]
105    InvalidImageWidth {
106        /// The width of the image being validated.
107        width: u32,
108        /// The number of characters per line in the layout.
109        per_line_character_count: u32,
110    },
111
112    /// The image height does not evenly divide the number of lines.
113    ///
114    /// This error occurs when the height of the provided image is not a
115    /// multiple of the number of lines in the layout.
116    #[error("Image height {height} is not an exact multiple of line count {line_count}.")]
117    InvalidImageHeight {
118        /// The height of the image being validated.
119        height: u32,
120        /// The number of lines in the layout.
121        line_count: u32,
122    },
123
124    /// A repeated character was found in an `Automatic` layout string.
125    ///
126    /// This error occurs when the same character appears multiple times in the
127    /// layout string, leading to conflicting placement definitions.
128    #[error(
129        "The character '{character}' appears more than once. The second appearance is in the \
130        layout string at row {row}, column {column}."
131    )]
132    AutomaticRepeatedCharacter {
133        /// The row in the layout string where the repeated character is
134        /// located.
135        row: usize,
136        /// The column in the layout string where the repeated character is
137        /// located.
138        column: usize,
139        /// The character that was repeated in the layout string.
140        character: char,
141    },
142}

144impl ImageFontLayout {
145    /// Given the image size, returns a map from each codepoint to its location.
146    #[expect(
147        clippy::cast_possible_truncation,
148        reason = "while usize can hold more data than u32, we're working on a number here that \
149        should be substantially smaller than even u32's capacity"
150    )]
151    fn into_character_rect_map(
152        self,
153        size: UVec2,
154    ) -> Result<HashMap<char, URect>, ImageFontLayoutValidationError> {
155        match self {
156            ImageFontLayout::Automatic(str) => {
157                // trim() removes whitespace, which is not what we want!
158                let str = str
159                    .trim_start_matches(['\r', '\n'])
160                    .trim_end_matches(['\r', '\n']);
161                #[expect(
162                    clippy::expect_used,
163                    reason = "this intentionally panics on an empty string. Should never happen as \
164                    the ImageFontLayout should always have been validated before this method gets \
165                    called"
166                )]
167                let max_chars_per_line = str
168                    .lines()
169                    // important: *not* l.len()
170                    .map(|line| line.chars().count())
171                    .max()
172                    .expect("can't create character map from an empty string")
173                    as u32;
174
175                if !size.x.is_multiple_of(max_chars_per_line) {
176                    return Err(ImageFontLayoutValidationError::InvalidImageWidth {
177                        width: size.x,
178                        per_line_character_count: max_chars_per_line,
179                    });
180                }
181                let line_count = str.lines().count() as u32;
182                if !size.y.is_multiple_of(line_count) {
183                    return Err(ImageFontLayoutValidationError::InvalidImageHeight {
184                        height: size.y,
185                        line_count,
186                    });
187                }
188
189                let mut rect_map =
190                    HashMap::with_capacity((max_chars_per_line * line_count) as usize);
191
192                let rect_width = size.x / max_chars_per_line;
193                let rect_height = size.y / line_count;
194
195                for (row, line) in str.lines().enumerate() {
196                    for (column, character) in line.chars().enumerate() {
197                        let rect = URect::new(
198                            rect_width * column as u32,
199                            rect_height * row as u32,
200                            rect_width * (column + 1) as u32,
201                            rect_height * (row + 1) as u32,
202                        );
203                        if rect_map.insert(character, rect).is_some() {
204                            return Err(
205                                ImageFontLayoutValidationError::AutomaticRepeatedCharacter {
206                                    row,
207                                    column,
208                                    character,
209                                },
210                            );
211                        }
212                    }
213                }
214
215                Ok(rect_map)
216            }
217            ImageFontLayout::ManualMonospace { size, coords } => Ok(coords
218                .into_iter()
219                .map(|(character, top_left)| {
220                    (character, URect::from_corners(top_left, size + top_left))
221                })
222                .collect()),
223            ImageFontLayout::Manual(urect_map) => Ok(urect_map),
224        }
225    }
226}

228/// On-disk representation of an [`ImageFont`], optimized to make it easy for
229/// humans to write these. See the docs for [`ImageFontLayout`]'s variants for
230/// information on how to write the syntax, or [the example font's RON asset].
231///
232/// [the example font's RON asset](https://github.com/ilyvion/bevy_image_font/blob/main/assets/example_font.image_font.ron)
233#[derive(Debug, Serialize, Deserialize)]
234pub struct ImageFontDescriptor {
235    /// The path to the image file containing the font glyphs, relative to the
236    /// RON file. This should be a valid path to a texture file that can be
237    /// loaded by the asset system.
238    image: Utf8PathBuf,
239
240    /// The layout description of the font, specifying how characters map to
241    /// regions within the image. This can use any of the variants provided
242    /// by [`ImageFontLayout`], allowing flexible configuration.
243    layout: ImageFontLayout,
244}
245
246/// Errors that can show up during validation.
247#[derive(Debug, Error)]
248#[non_exhaustive]
249pub enum ImageFontDescriptorValidationError {
250    /// The image path provided is empty.
251    #[error("Image path is empty.")]
252    EmptyImagePath,
253
254    /// The layout string used for automatic character placement is empty.
255    /// This error occurs when no characters are defined in the automatic layout
256    /// string.
257    #[error("Automatic layout string is empty.")]
258    EmptyLayoutString,
259}
260
261impl ImageFontDescriptor {
262    /// Creates a new `ImageFontDescriptor` instance with the provided image
263    /// path and font layout, performing validation to ensure the descriptor
264    /// is valid.
265    ///
266    /// # Parameters
267    /// - `image`: The path to the image file containing the font glyphs,
268    ///   relative to the RON file that described the font. This should be a
269    ///   valid path to a texture file that can be loaded by the asset system.
270    /// - `layout`: The layout description of the font, specifying how
271    ///   characters map to regions within the image. See [`ImageFontLayout`]
272    ///   for more details about the available layout configurations.
273    ///
274    /// # Returns
275    /// A new `ImageFontDescriptor` instance if validation succeeds.
276    ///
277    /// # Errors
278    /// Returns an [`ImageFontDescriptorValidationError`] if the provided values
279    /// do not pass validation.
280    pub fn new(
281        image: Utf8PathBuf,
282        layout: ImageFontLayout,
283    ) -> Result<Self, ImageFontDescriptorValidationError> {
284        let value = Self { image, layout };
285        value.validate()?;
286        Ok(value)
287    }
288
289    /// Validates the `ImageFontDescriptor` struct to ensure all required fields
290    /// are populated.
291    ///
292    /// # Errors
293    ///   - `ImageFontLoadError::EmptyImagePath` if the `image` path is empty.
294    ///   - `ImageFontLoadError::EmptyLayoutString` if the `layout` string for
295    ///     `Automatic` is empty.
296    fn validate(&self) -> Result<(), ImageFontDescriptorValidationError> {
297        if self.image.as_str().trim().is_empty() {
298            return Err(ImageFontDescriptorValidationError::EmptyImagePath);
299        }
300        if matches!(self.layout, ImageFontLayout::Automatic(ref layout) if layout.trim().is_empty())
301        {
302            return Err(ImageFontDescriptorValidationError::EmptyLayoutString);
303        }
304        Ok(())
305    }
306
307    /// Gets the path to the image file containing the font glyphs.
308    ///
309    /// This is the value of the `image` field. The path is relative to the
310    /// RON file and should point to a valid texture file.
311    ///
312    /// # Returns
313    /// A reference to the `Utf8PathBuf` containing the image file path.
314    #[must_use]
315    pub fn image(&self) -> &Utf8Path {
316        &self.image
317    }
318
319    /// Gets the layout description of the font.
320    ///
321    /// This is the value of the `layout` field, which specifies how characters
322    /// map to regions within the image. See [`ImageFontLayout`] for details
323    /// about the available variants.
324    ///
325    /// # Returns
326    /// A reference to the `ImageFontLayout` describing the font layout.
327    #[must_use]
328    pub fn layout(&self) -> &ImageFontLayout {
329        &self.layout
330    }
331}

337/// Errors that can show up during font loading.
338#[derive(Debug, Error)]
339#[non_exhaustive]
340pub enum ImageFontLoadError {
341    /// Parsing the on-disk representation of the font failed. This typically
342    /// indicates a syntax or formatting error in the RON file.
343    #[error("couldn't parse on-disk representation: {0}")]
344    ParseFailure(#[from] SpannedError),
345
346    /// A validation error occurred on the `ImageFontDescriptor`. Inspect the
347    /// value of the inner error for details.
348    #[error("Font descriptor is invalid: {0}")]
349    DescriptorValidationError(#[from] ImageFontDescriptorValidationError),
350
351    /// A validation error occurred on the `ImageFontLayout`. Inspect the
352    /// value of the inner error for details.
353    #[error("Font layout is invalid: {0}")]
354    LayoutValidationError(#[from] ImageFontLayoutValidationError),
355
356    /// An I/O error occurred while loading the image font. This might happen
357    /// if the file cannot be accessed, is missing, or is corrupted.
358    #[error("i/o error when loading image font: {0}")]
359    Io(#[from] IoError),
360
361    /// Failed to load an asset directly. This is usually caused by an error
362    /// in the asset pipeline or a missing dependency.
363    #[error("failed to load asset: {0}")]
364    LoadDirect(Box<LoadDirectError>),
365
366    /// The path provided for the font's image was not loaded as an image. This
367    /// may occur if the file is in an unsupported format or if the path is
368    /// incorrect.
369    #[error("Path does not point to a valid image file: {0}")]
370    NotAnImage(Utf8PathBuf),
371
372    /// The path provided for the font's image was not loaded as an image. This
373    /// may occur if the file is in an unsupported format or if the path is
374    /// incorrect.
375    #[error("Path is not valid UTF-8: {0:?}")]
376    InvalidPath(FromPathError),
377
378    /// The asset path has no parent directory.
379    #[error("Asset path has no parent directory")]
380    MissingParentPath,
381}
382
383impl From<LoadDirectError> for ImageFontLoadError {
384    #[inline]
385    fn from(value: LoadDirectError) -> Self {
386        Self::LoadDirect(Box::new(value))
387    }
388}

390/// Configuration settings for the `ImageFontLoader`.
391#[derive(Debug, serde::Deserialize, serde::Serialize)]
392#[non_exhaustive]
393pub struct ImageFontLoaderSettings {
394    /// The [`ImageSampler`] to use during font image rendering. Determines
395    /// how the font's texture is sampled when scaling or transforming it.
396    ///
397    /// The default is `nearest`, which scales the image without blurring,
398    /// preserving a crisp, pixelated appearance. This is usually ideal for
399    /// pixel-art fonts.
400    pub image_sampler: ImageSampler,
401}
402
403impl Default for ImageFontLoaderSettings {
404    fn default() -> Self {
405        Self {
406            image_sampler: ImageSampler::Descriptor(ImageSamplerDescriptor::nearest()),
407        }
408    }
409}

411impl AssetLoader for ImageFontLoader {
412    type Asset = ImageFont;
413
414    type Settings = ImageFontLoaderSettings;
415
416    type Error = ImageFontLoadError;
417
418    // NOTE: Until I or someone else thinks of a way to reliably run `AssetLoaders`
419    //       in a unit test, parts of this method will unfortunately remain
420    //       uncovered by tests.
421    async fn load(
422        &self,
423        reader: &mut dyn Reader,
424        settings: &Self::Settings,
425        load_context: &mut LoadContext<'_>,
426    ) -> Result<Self::Asset, Self::Error> {
427        let font_descriptor = read_and_validate_font_descriptor(reader).await?;
428
429        // need the image loaded immediately because we need its size
430        let image_path = load_context
431            .path()
432            .parent()
433            .ok_or(ImageFontLoadError::MissingParentPath)?
434            .join(font_descriptor.image());
435        let Some(mut image) = load_context
436            .loader()
437            .immediate()
438            .with_unknown_type()
439            .load(image_path.as_path())
440            .await?
441            .take::<Image>()
442        else {
443            let path = match Utf8PathBuf::try_from(image_path) {
444                Ok(path) => path,
445                Err(error) => return Err(ImageFontLoadError::InvalidPath(error.from_path_error())),
446            };
447            return Err(ImageFontLoadError::NotAnImage(path));
448        };
449
450        image.sampler = settings.image_sampler.clone();
451        let size = image.size();
452
453        let (atlas_character_map, layout) =
454            descriptor_to_character_map_and_layout(font_descriptor, size)?;
455
456        let image_handle = load_context.add_labeled_asset(String::from("texture"), image);
457        let layout_handle = load_context.add_labeled_asset(String::from("layout"), layout);
458
459        let image_font = ImageFont::new(
460            vec![image_handle],
461            atlas_character_map,
462            vec![layout_handle],
463            settings.image_sampler.clone(),
464        );
465        Ok(image_font)
466    }
467
468    fn extensions(&self) -> &[&str] {
469        &["image_font.ron"]
470    }
471}

473/// Reads and validates an `ImageFontDescriptor` from a reader.
474///
475/// This function reads the entirety of the data provided by the `reader`,
476/// deserializes it into an `ImageFontDescriptor`, and performs validation
477/// to ensure the descriptor is valid.
478///
479/// # Parameters
480/// - `reader`: A mutable reference to an object implementing the [`Reader`]
481///   trait. This reader provides the serialized data for the font descriptor.
482///
483/// # Returns
484/// A `Result` containing either a valid `ImageFontDescriptor` or an error if
485/// reading, deserialization, or validation fails.
486///
487/// # Errors
488/// Returns an error in the following cases:
489/// - If reading from the `reader` fails.
490/// - If the data cannot be deserialized into an `ImageFontDescriptor`.
491/// - If the resulting `ImageFontDescriptor` does not pass validation.
492async fn read_and_validate_font_descriptor(
493    reader: &mut dyn Reader,
494) -> Result<ImageFontDescriptor, ImageFontLoadError> {
495    // Read data
496    let mut data = Vec::new();
497    reader.read_to_end(&mut data).await?;
498
499    // Deserialize into ImageFontDescriptor and validate
500    let font_descriptor: ImageFontDescriptor = ron::de::from_bytes(&data)?;
501    font_descriptor.validate()?;
502
503    Ok(font_descriptor)
504}

506/// Converts an `ImageFontDescriptor` into a character map and texture atlas
507/// layout.
508///
509/// This function processes the given `ImageFontDescriptor` to generate a
510/// character-to-index map and a [`TextureAtlasLayout`], based on the provided
511/// image size. It uses the descriptor's layout information to map characters to
512/// specific regions within the texture atlas.
513///
514/// # Parameters
515/// - `font_descriptor`: The `ImageFontDescriptor` containing the layout.
516/// - `image_size`: A [`UVec2`] representing the dimensions of the image
517///   containing the font glyphs.
518///
519/// # Returns
520/// A tuple where
521/// - the first element is a `HashMap<char, usize>` mapping characters to
522///   indices in the texture atlas.
523/// - the second element is a [`TextureAtlasLayout`] describing the texture
524///   atlas layout.
525///
526/// # Errors
527/// This function will return an [`ImageFontLoadError`] in the following cases:
528/// - If there are any validation errors in the layout. See
529///   [`ImageFontLayoutValidationError`] for details.
530fn descriptor_to_character_map_and_layout(
531    font_descriptor: ImageFontDescriptor,
532    image_size: UVec2,
533) -> Result<(HashMap<char, ImageFontCharacter>, TextureAtlasLayout), ImageFontLoadError> {
534    let rect_character_map = font_descriptor.layout.into_character_rect_map(image_size)?;
535    let (atlas_character_map, layout) =
536        ImageFont::mapped_atlas_layout_from_char_map(0, image_size, rect_character_map.into_iter());
537    Ok((atlas_character_map, layout))
538}
