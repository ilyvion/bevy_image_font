1//! This module provides functionality for rendering text as images in the Bevy
2//! engine, utilizing custom image fonts. It includes components for
3//! pre-rendering text for both in-world and UI contexts, as well as systems to
4//! update and render the text when changes occur.
5//!
6//! Key Features:
7//! - `ImageFontPreRenderedText` and `ImageFontPreRenderedUiText` components for
8//!   in-world and UI text rendering, respectively.
9//! - Systems for rendering text updates to `Sprite` or `ImageNode` components
10//!   dynamically.
11//! - Integrates with the `image` crate for low-level image manipulation.

87/// System that renders each [`ImageFontText`] into its [`Sprite`]. This system
88/// only runs when the `ImageFontText` changes.
89pub fn render_text_to_sprite(
90    mut query: Query<(&ImageFontText, &mut Sprite), Changed<ImageFontText>>,
91    image_fonts: Res<Assets<ImageFont>>,
92    mut images: ResMut<Assets<Image>>,
93    layouts: Res<Assets<TextureAtlasLayout>>,
94) {
95    render_text_to_image_holder(
96        query
97            .iter_mut()
98            .map(|(image_font_text, sprite)| (image_font_text, sprite.into_inner())),
99        &image_fonts,
100        &mut images,
101        &layouts,
102    );
103}

124/// Renders text into images and assigns the resulting image handles to their
125/// holders.
126///
127/// This function is designed to work with any type that implements
128/// [`ImageHandleHolder`], allowing it to be used with multiple components, such
129/// as sprites and UI elements.
130///
131/// # Parameters
132/// - `font_text_to_image_iter`: An iterator over pairs of [`ImageFontText`] and
133///   mutable references to objects implementing [`ImageHandleHolder`]. Each
134///   item in the iterator represents a text-to-image mapping to be rendered.
135/// - `image_fonts`: A reference to the font assets used for rendering.
136/// - `images`: A mutable reference to the collection of image assets. This is
137///   used to store the newly rendered images.
138/// - `layouts`: A reference to the collection of texture atlas assets.
139///
140/// The function iterates over the provided items, renders the text for each
141/// [`ImageFontText`], and updates the corresponding [`ImageHandleHolder`] with
142/// the handle to the newly created image.
143///
144/// # Errors
145/// If text rendering fails for an item (e.g., due to missing font assets
146/// or invalid texture layouts), an error message is logged, and the
147/// corresponding holder is not updated.
148fn render_text_to_image_holder<'borrow>(
149    font_text_to_image_iter: impl Iterator<
150        Item = (
151            &'borrow ImageFontText,
152            &'borrow mut (impl ImageHandleHolder + 'borrow),
153        ),
154    >,
155    image_fonts: &Assets<ImageFont>,
156    images: &mut Assets<Image>,
157    layouts: &Assets<TextureAtlasLayout>,
158) {
159    for (image_font_text, image_handle_holder) in font_text_to_image_iter {
160        debug!("Rendering [{}]", image_font_text.text);
161        match render_text_to_image(image_font_text, image_fonts, images, layouts) {
162            Ok(image) => {
163                image_handle_holder.set_image_handle(images.add(image));
164            }
165            Err(error) => {
166                error!(
167                    "Error when rendering image font text {:?}: {}",
168                    image_font_text, error
169                );
170            }
171        }
172    }
173}

175/// Renders the text from an [`ImageFontText`] into a single image.
176///
177/// This function takes a reference to an [`ImageFontText`] component and
178/// generates an image representation of the text. It applies font-specific
179/// filtering, determines appropriate character placements, and composites the
180/// final output.
181///
182/// # Parameters
183/// - `image_font_text`: The text to render, along with its associated font.
184/// - `image_fonts`: The collection of available font assets.
185/// - `images`: The collection of image assets used to retrieve font textures.
186/// - `layouts`: The texture atlas layouts defining character positioning.
187///
188/// # Returns
189/// A [`Result<Image, ImageFontRenderError>`] containing the generated image if
190/// successful, or an error if rendering fails (e.g., due to missing assets).
191///
192/// # Behavior
193/// - If the text is empty, a **1x1 transparent image** is returned to avoid
194///   invalid texture sizes.
195/// - The function leverages [`RenderContext`] to compute character positions
196///   and generate the image.
197#[expect(
198    clippy::cast_possible_truncation,
199    clippy::cast_sign_loss,
200    reason = "numbers are always positive and small enough"
201)]
202fn render_text_to_image(
203    image_font_text: &ImageFontText,
204    image_fonts: &Assets<ImageFont>,
205    images: &Assets<Image>,
206    layouts: &Assets<TextureAtlasLayout>,
207) -> Result<Image, ImageFontRenderError> {
208    let (image_font, textures, mut render_context) =
209        prepare_context(image_font_text, image_fonts, images, layouts)?;
210
211    if render_context.text().is_empty() {
212        // Can't make a 0x0 image, so make a 1x1 transparent black pixel
213        return Ok(Image::new(
214            Extent3d {
215                width: 1,
216                height: 1,
217                depth_or_array_layers: 1,
218            },
219            TextureDimension::D2,
220            vec![0, 0, 0, 0],
221            TextureFormat::Rgba8UnormSrgb,
222            RenderAssetUsages::RENDER_WORLD,
223        ));
224    }
225
226    let width = render_context.text_width() as u32;
227    let height = render_context.max_height();
228    let mut output_image = image::RgbaImage::new(width, height);
229    let font_textures: Vec<ImageBuffer<Rgba<u8>, _>> = textures
230        .iter()
231        .map(|texture| {
232            let data = texture.data.as_ref()?;
233            ImageBuffer::from_raw(texture.width(), texture.height(), data.as_slice())
234        })
235        .collect::<Option<_>>()
236        .ok_or(ImageFontRenderError::UnknownError)?;
237
238    render_glyphs_to_image(
239        layouts,
240        image_font,
241        &render_context,
242        &mut output_image,
243        font_textures,
244    )?;
245
246    #[expect(
247        clippy::cast_possible_truncation,
248        clippy::cast_sign_loss,
249        reason = "the magnitude of the numbers we're working on here are too small to lose anything"
250    )]
251    if let Some(font_height) = image_font_text.font_height {
252        render_context.render_config.apply_scaling = true;
253        let scaled_width = render_context.text_width();
254
255        output_image = imageops::resize(
256            &output_image,
257            scaled_width as u32,
258            font_height as u32,
259            FilterType::Nearest,
260        );
261    }
262
263    let mut bevy_image = Image::new(
264        Extent3d {
265            width: output_image.width(),
266            height: output_image.height(),
267            depth_or_array_layers: 1,
268        },
269        TextureDimension::D2,
270        output_image.into_vec(),
271        TextureFormat::Rgba8UnormSrgb,
272        RenderAssetUsages::RENDER_WORLD,
273    );
274    bevy_image.sampler = ImageSampler::nearest();
275
276    Ok(bevy_image)
277}

279/// Renders each glyph of the filtered text into the output image buffer.
280///
281/// This function iterates over the filtered characters in the text, copying
282/// each glyph from the appropriate font texture into the output image at the
283/// correct position. It uses the render context to determine glyph placement,
284/// color, and texture atlas information. The function also handles updating the
285/// x-position for each glyph, including any kerning or advance adjustments, by
286/// delegating to the render context's transform logic.
287///
288/// # Parameters
289/// - `layouts`: The collection of texture atlas layouts for glyph positioning.
290/// - `image_font`: The font asset containing glyph metadata and character maps.
291/// - `render_context`: The context providing filtered text and rendering
292///   configuration.
293/// - `output_image`: The image buffer to which glyphs are rendered.
294/// - `font_textures`: The set of font texture images, one per atlas page.
295///
296/// # Errors
297/// Returns an error if copying a glyph from the font texture to the output
298/// image fails.
299#[expect(
300    clippy::cast_possible_truncation,
301    clippy::cast_sign_loss,
302    reason = "numbers are always positive and small enough"
303)]
304fn render_glyphs_to_image(
305    layouts: &Assets<TextureAtlasLayout>,
306    image_font: &ImageFont,
307    render_context: &RenderContext<'_>,
308    output_image: &mut ImageBuffer<Rgba<u8>, Vec<u8>>,
309    font_textures: Vec<ImageBuffer<Rgba<u8>, &[u8]>>,
310) -> Result<(), ImageFontRenderError> {
311    let mut x_pos = 0.0;
312    let mut texture_atlas = render_context.font_texture_atlas(' ');
313    let mut color = Color::default();
314
315    for (character, next_character) in render_context
316        .text()
317        .filtered_chars()
318        .map(Some)
319        .chain(iter::once(None))
320        .tuple_windows()
321    {
322        #[expect(
323            clippy::expect_used,
324            reason = "tuple_windows() guarantees that only the last next_character is None"
325        )]
326        let character = character.expect("Only the last character can be None");
327
328        let image_font_character = &image_font.atlas_character_map[&character];
329        render_context.update_render_values(character, &mut texture_atlas, &mut color);
330
331        #[expect(
332            clippy::expect_used,
333            reason = "we're using `filtered_chars()` which has only valid characters"
334        )]
335        let rect = texture_atlas
336            .texture_rect(layouts)
337            .expect("`filtered_chars()` guarantees valid characters");
338
339        output_image.copy_from(
340            &*font_textures[image_font_character.page_index].view(
341                rect.min.x,
342                rect.min.y,
343                rect.width(),
344                rect.height(),
345            ),
346            x_pos as u32,
347            0,
348        )?;
349
350        // Let `transform()` handle x-position updates
351        render_context.transform(&mut x_pos, character, next_character);
352    }
353    Ok(())
354}

356/// Prepares the rendering context for image font text rendering.
357///
358/// This function retrieves the required font asset, associated textures, and
359/// constructs a `RenderContext` for the given `ImageFontText`. It returns all
360/// the necessary data to render the text as an image, including the font,
361/// textures, and context with rendering configuration.
362///
363/// # Parameters
364/// - `image_font_text`: The text and font information to render.
365/// - `image_fonts`: The collection of available font assets.
366/// - `images`: The collection of image assets used to retrieve font textures.
367/// - `layouts`: The texture atlas layouts defining character positioning.
368///
369/// # Returns
370/// A tuple containing:
371/// - A reference to the `ImageFont` asset.
372/// - A vector of references to the font's texture images.
373/// - The constructed `RenderContext` for rendering.
374///
375/// # Errors
376/// Returns an error if the font asset or required texture assets are missing.
377fn prepare_context<'assets, 'image>(
378    image_font_text: &'assets ImageFontText,
379    image_fonts: &'assets Assets<ImageFont>,
380    images: &'image Assets<Image>,
381    layouts: &'assets Assets<TextureAtlasLayout>,
382) -> Result<
383    (
384        &'assets ImageFont,
385        Vec<&'image Image>,
386        RenderContext<'assets>,
387    ),
388    ImageFontRenderError,
389> {
390    let image_font = image_fonts
391        .get(&image_font_text.font)
392        .ok_or(ImageFontRenderError::MissingImageFontAsset)?;
393    let textures = image_font.textures(images);
394    let render_config = RenderConfig {
395        text_anchor: Anchor::Center,
396        offset_characters: false,
397        apply_scaling: false,
398        letter_spacing: 0.0,
399        scaling_mode: ImageFontScalingMode::Truncated,
400        color: Color::WHITE, // Currently unused for rendering to an image
401    };
402    let render_context = RenderContext::new(image_font, image_font_text, render_config, layouts)
403        .ok_or(ImageFontRenderError::MissingTextureAsset)?;
404    Ok((image_font, textures, render_context))
405}

407/// Errors that can occur during the rendering of an `ImageFont`.
408#[derive(Debug, Error)]
409#[non_exhaustive]
410pub enum ImageFontRenderError {
411    /// The image could not be converted to a `DynamicImage`. This typically
412    /// indicates an issue with the underlying image data or its format.
413    #[error("failed to convert image to DynamicImage: {0}")]
414    ImageConversion(String),
415
416    /// The `ImageFont` asset required for rendering was not loaded. Ensure
417    /// that the font asset is correctly loaded into the Bevy app. This should
418    /// happend automatically when using the `AssetLoader` to load the font
419    /// asset.
420    #[error("ImageFont asset not loaded")]
421    MissingImageFontAsset,
422
423    /// The texture asset associated with the `ImageFont` was not loaded. This
424    /// could indicate an issue with the asset pipeline or a missing dependency.
425    /// This should happend automatically when using the `AssetLoader` to
426    /// load the font asset.
427    #[error("Font texture asset not loaded")]
428    MissingTextureAsset,
429
430    /// An unspecified internal error occurred during rendering. This may
431    /// indicate a bug or unexpected state in the rendering system.
432    #[error("internal error")]
433    UnknownError,
434
435    /// Failed to copy a character from the source font image texture to the
436    /// target rendered text sprite image texture. This error typically occurs
437    /// when there is an issue with the image data or the copying process.
438    #[error("failed to copy from atlas: {0}")]
439    CopyFailure(#[from] ImageError),
440}

442/// A helper trait that represents types that can hold an image handle.
443///
444/// This is used to abstract over different components (e.g., [`Sprite`] and
445/// [`ImageNode`]) that need to store a handle to an image rendered from text.
446trait ImageHandleHolder {
447    /// Sets the handle for the image that this holder represents.
448    ///
449    /// This method is called after rendering text into an image
450    /// to update the holder with the new image handle.
451    ///
452    /// # Parameters
453    /// - `image`: The handle to the newly rendered image.
454    fn set_image_handle(&mut self, image: Handle<Image>);
455}
456
457impl ImageHandleHolder for Sprite {
458    fn set_image_handle(&mut self, image: Handle<Image>) {
459        self.image = image;
460    }
461}
462
463#[cfg(feature = "ui")]
464impl ImageHandleHolder for ImageNode {
465    fn set_image_handle(&mut self, image: Handle<Image>) {
466        self.image = image;
467    }
