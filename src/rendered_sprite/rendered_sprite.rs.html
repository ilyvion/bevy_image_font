1//! Demonstrates rendering image font text at both its 'native' height and a
2//! scaled-up height.
3
4#![expect(
5    clippy::mod_module_files,
6    reason = "if present as common.rs, cargo thinks it's an example binary"
7)]
8
9use bevy::prelude::*;
10use bevy::sprite::Anchor;
11use bevy_asset_loader::prelude::AssetCollectionApp as _;
12use bevy_image_font::rendered::ImageFontPreRenderedText;
13use bevy_image_font::{ImageFontPlugin, ImageFontText};
14
15use crate::common::{DemoAssets, FONT_WIDTH, TEXT};
16
17mod common;
18
19fn main() {
20    App::new()
21        .add_plugins((DefaultPlugins, ImageFontPlugin))
22        .init_collection::<DemoAssets>()
23        .add_systems(Startup, spawn_text)
24        .insert_resource(ClearColor(Color::srgb(0.2, 0.2, 0.2)))
25        .run();
26}

28/// Spawns the text entities for the example.
29///
30/// This system creates two text entities:
31/// 1. A text entity rendered at a scaled-up height for demonstration purposes.
32/// 2. A text entity rendered at its native height, demonstrating pixel-perfect
33///    alignment.
34///
35/// The first entity shows how to shift the position by 0.5 pixels to align with
36/// pixel boundaries. The second demonstrates the use of an anchor to achieve
37/// proper alignment.
38fn spawn_text(mut commands: Commands, assets: Res<DemoAssets>) {
39    commands.spawn(Camera2d);
40
41    commands.spawn((
42        ImageFontPreRenderedText::default(),
43        ImageFontText::default()
44            .text(TEXT)
45            .font(assets.example.clone())
46            .font_height(36.0),
47        // Our font is 45 pixels wide per character, and with an odd number of characters, the
48        // text aligns to the middle of a pixel, causing imperfect rendering. Shifting the
49        // position by 0.5 pixels ensures alignment to pixel boundaries, resulting in crisp,
50        // pixel-perfect rendering.
51        Transform::from_translation(Vec3::new(0.5, 0., 0.)),
52    ));
53    commands.spawn((
54        ImageFontPreRenderedText::default(),
55        ImageFontText::default()
56            .text(TEXT)
57            .font(assets.example.clone()),
58        // Instead of shifting the character by 0.5 pixels when the text lands in the middle of
59        // a pixel, we can anchor the sprite to an edge and move it by a whole number of pixels.
60        // To center it with the text above, we shift it left by half its width.
61        Sprite {
62            anchor: Anchor::CenterLeft,
63            ..default()
64        },
65        #[expect(
66            clippy::cast_precision_loss,
67            reason = "the magnitude of the numbers we're working on here are too small to lose \
68            anything"
69        )]
70        Transform::from_translation(Vec3::new(
71            -((TEXT.chars().count() * FONT_WIDTH / 2) as f32),
72            40.,
73            0.,
74        )),
75    ));
76}
