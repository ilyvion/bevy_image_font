1//! Shared utilities and constants for example binaries.
2//!
3//! This module provides common functionality and configuration shared across
4//! the example binaries for the project. It includes reusable constants,
5//! such as default text and color palettes, that are utilized to ensure
6//! consistency and reduce duplication across examples.
7//!
8//! # Key Features
9//! - **Default Text:** Includes a pangram for rendering demonstrations and
10//!   testing.
11//! - **Font Configuration:** Provides the font width of the example font.
12//! - **Rainbow Colors:** Supplies a palette of colors for visual styling in
13//!   examples.
14//!
15//! # Usage
16//! This module is intended for internal use by example binaries. It reduces
17//! redundancy by centralizing common assets and configurations. Depending on
18//! the active feature set, some utilities may not be used in certain examples.
19
20#![allow(
21    dead_code,
22    reason = "private utility code that, depending on the activated feature set, \
23    will sometimes be missing uses"
24)]

115/// Gizmos related example code
116#[cfg(all(feature = "gizmos", feature = "atlas_sprites"))]
117pub(crate) mod gizmos {
118    use bevy::{
119        ecs::system::{Res, ResMut},
120        gizmos::config::GizmoConfigStore,
121        input::{ButtonInput, keyboard::KeyCode},
122    };
123    use bevy_image_font::atlas_sprites::gizmos::AtlasSpritesGizmoConfigGroup;
124
125    //use super::{ButtonInput, GizmoConfigStore, KeyCode, Res, ResMut};
126
127    /// Configures default gizmo rendering settings.
128    ///
129    /// This function initializes the default visibility settings for
130    /// text-related gizmos, enabling anchor points and bounding box
131    /// rendering by default.
132    ///
133    /// # Default Behavior
134    /// - **Text anchor points**: Enabled (`render_text_anchor_point = true`).
135    /// - **Character anchor points**: Enabled (`render_character_anchor_point =
136    ///   true`).
137    /// - **Character bounding boxes**: Enabled (`render_character_box = true`).
138    ///
139    /// This function is called at startup to ensure gizmos are visible by
140    /// default.
141    pub(crate) fn configure_gizmo_defaults(mut store: ResMut<'_, GizmoConfigStore>) {
142        let (_, atlas_sprites_config) = store.config_mut::<AtlasSpritesGizmoConfigGroup>();
143        atlas_sprites_config.render_text_anchor_point = true;
144        atlas_sprites_config.render_character_anchor_point = true;
145        atlas_sprites_config.render_character_box = true;
146    }
147
148    /// Handles keyboard input to toggle gizmo rendering settings.
149    ///
150    /// This function listens for key presses and updates the gizmo
151    /// configuration to enable or disable various debug visuals in real
152    /// time.
153    ///
154    /// # Parameters
155    /// - `input`: Reference to [`ButtonInput<KeyCode>`], used to detect key
156    ///   presses.
157    /// - `store`: Mutable reference to the [`GizmoConfigStore`] to update
158    ///   settings.
159    ///
160    /// # Keyboard Shortcuts
161    /// - **`G`**: Toggles all gizmo rendering on/off.
162    /// - **`A`**: Toggles character anchor point gizmos.
163    /// - **`B`**: Toggles character bounding box gizmos.
164    /// - **`T`**: Toggles text anchor point gizmos.
165    pub(crate) fn toggle_gizmos(
166        input: Res<ButtonInput<KeyCode>>,
167        mut store: ResMut<GizmoConfigStore>,
168    ) {
169        let (config, atlas_sprites_config) = store.config_mut::<AtlasSpritesGizmoConfigGroup>();
170        if input.just_pressed(KeyCode::KeyG) {
171            config.enabled = !config.enabled;
172            println!("Gizmos enabled: {}", config.enabled);
173        }
174        if input.just_pressed(KeyCode::KeyA) {
175            atlas_sprites_config.render_character_anchor_point =
176                !atlas_sprites_config.render_character_anchor_point;
177            println!(
178                "Character anchor point gizmo enabled: {}",
179                atlas_sprites_config.render_character_anchor_point
180            );
181        }
182        if input.just_pressed(KeyCode::KeyB) {
183            atlas_sprites_config.render_character_box = !atlas_sprites_config.render_character_box;
184            println!(
185                "Character box gizmo enabled: {}",
186                atlas_sprites_config.render_character_box
187            );
188        }
189        if input.just_pressed(KeyCode::KeyT) {
190            atlas_sprites_config.render_text_anchor_point =
191                !atlas_sprites_config.render_text_anchor_point;
192            println!(
193                "Text anchor gizmo enabled: {}",
194                atlas_sprites_config.render_text_anchor_point
195            );
196        }
197    }
