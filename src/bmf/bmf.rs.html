1//! # Bevy Image Font Example
2//!
3//! This example demonstrates how to use `bevy_image_font` to render text
4//! with bitmap fonts in a Bevy application. It loads multiple font formats
5//! and displays multilingual pangrams to test font rendering across different
6//! scripts and writing systems.
7//!
8//! ## Highlight: Pangrams for Multilingual Script Testing
9//! A **pangram** is a sentence that contains every letter of an alphabet at
10//! least once. This example renders pangrams from various languages and
11//! scripts, including:
12//! - Latin-based languages (English, French, German, etc.).
13//! - Cyrillic (Russian).
14//! - Greek.
15//! - Hebrew and Arabic (right-to-left scripts).
16//! - Thai and Armenian.
17//!
18//! By displaying pangrams, this example provides a visually rich way to test
19//! font rendering, spacing, and character coverage across different writing
20//! systems.
21
22#![expect(
23    clippy::mod_module_files,
24    reason = "if present as common.rs, cargo thinks it's an example binary"
25)]
26
27use bevy::log::tracing_subscriber::fmt::format::FmtSpan;
28use bevy::log::tracing_subscriber::fmt::layer;
29use bevy::log::tracing_subscriber::Layer as _;
30use bevy::log::LogPlugin;
31use bevy::prelude::*;
32use bevy_asset_loader::prelude::AssetCollectionApp as _;
33use bevy_image_font::atlas_sprites::ImageFontSpriteText;
34use bevy_image_font::{ImageFontPlugin, ImageFontText};
35
36use crate::common::{DemoAssets, PANGRAMS};
37
38mod common;
39
40fn main() {
41    let log_plugin = LogPlugin {
42        custom_layer: |_| {
43            //let log_file = File::create("bmf.log").unwrap();
44            Some(
45                layer()
46                    .with_span_events(FmtSpan::ACTIVE)
47                    .with_file(true)
48                    .with_line_number(true)
49                    .with_ansi(false)
50                    .pretty()
51                    .boxed(),
52            )
53        },
54        ..default()
55    };
56    App::new()
57        .add_plugins((
58            DefaultPlugins
59                .set(ImagePlugin::default_nearest())
60                .set(log_plugin),
61            ImageFontPlugin,
62        ))
63        .init_collection::<DemoAssets>()
64        .add_systems(Startup, spawn_text)
65        .insert_resource(ClearColor(Color::srgb(0.2, 0.2, 0.2)))
66        .run();
67}
68
69/// Spawns the text entities for the example.
70///
71/// This system creates two text entities:
72/// 1. A text entity rendered at a scaled height with animated colors.
73/// 2. A text entity rendered at its native height with animated content.
74fn spawn_text(mut commands: Commands, assets: Res<DemoAssets>) {
75    commands.spawn(Camera2d);
76
77    for (i, &pangram) in PANGRAMS.iter().enumerate() {
78        let font = match i % 3 {
79            0 => assets.bmf_txt.clone_weak(),
80            1 => assets.bmf_xml.clone_weak(),
81            2 => assets.bmf_bin.clone_weak(),
82            _ => unreachable!(),
83        };
84
85        commands.spawn((
86            ImageFontSpriteText::default(),
87            ImageFontText::default().text(pangram).font(font),
88            #[expect(
89                clippy::cast_precision_loss,
90                reason = "the magnitude of the numbers we're working on here are too small to lose \
91                anything"
92            )]
93            Transform::from_translation(Vec3::new(
94                0.,
95                PANGRAMS.len() as f32 * 20. + -(i as f32) * 40.,
96                0.,
97            )),
98        ));
99    }
100}
