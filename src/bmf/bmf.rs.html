1//! # Bevy Image Font Example
2//!
3//! This example demonstrates how to use `bevy_image_font` to render text
4//! with bitmap fonts in a Bevy application. It loads multiple font formats
5//! and displays multilingual pangrams to test font rendering across different
6//! scripts and writing systems.
7//!
8//! ## Highlight: Pangrams for Multilingual Script Testing
9//! A **pangram** is a sentence that contains every letter of an alphabet at
10//! least once. This example renders pangrams from various languages and
11//! scripts, including:
12//! - Latin-based languages (English, French, German, etc.).
13//! - Cyrillic (Russian).
14//! - Greek.
15//! - Hebrew and Arabic (right-to-left scripts).
16//! - Thai and Armenian.
17//!
18//! By displaying pangrams, this example provides a visually rich way to test
19//! font rendering, spacing, and character coverage across different writing
20//! systems.
21
22#![expect(
23    clippy::mod_module_files,
24    reason = "if present as common.rs, cargo thinks it's an example binary"
25)]
26
27use bevy::prelude::*;
28use bevy_asset_loader::prelude::AssetCollectionApp as _;
29#[cfg(feature = "gizmos")]
30use bevy_gizmos::config::GizmoConfigStore;
31use bevy_image_font::atlas_sprites::ImageFontSpriteText;
32use bevy_image_font::{ImageFontPlugin, ImageFontText};
33
34use crate::common::{DemoAssets, PANGRAMS};
35
36mod common;
37
38fn main() {
39    let mut app = App::new();
40
41    app.add_plugins((
42        DefaultPlugins.set(ImagePlugin::default_nearest()),
43        ImageFontPlugin,
44    ))
45    .init_collection::<DemoAssets>()
46    .add_systems(Startup, setup)
47    .insert_resource(ClearColor(Color::srgb(0.2, 0.2, 0.2)));
48
49    #[cfg(feature = "gizmos")]
50    app.add_systems(Update, common::gizmos::toggle_gizmos);
51
52    app.run();
53}
54
55/// Spawns the text entities for the example.
56///
57/// This system creates two text entities:
58/// 1. A text entity rendered at a scaled height with animated colors.
59/// 2. A text entity rendered at its native height with animated content.
60fn setup(
61    mut commands: Commands,
62    assets: Res<DemoAssets>,
63    #[cfg(feature = "gizmos")] store: ResMut<GizmoConfigStore>,
64) {
65    #[cfg(feature = "gizmos")]
66    common::gizmos::configure_gizmo_defaults(store);
67
68    commands.spawn(Camera2d);
69
70    for (i, &pangram) in PANGRAMS.iter().enumerate() {
71        let font = match i % 3 {
72            0 => assets.bmf_txt.clone_weak(),
73            1 => assets.bmf_xml.clone_weak(),
74            2 => assets.bmf_bin.clone_weak(),
75            _ => unreachable!(),
76        };
77
78        commands.spawn((
79            ImageFontSpriteText::default(),
80            ImageFontText::default().text(pangram).font(font),
81            #[expect(
82                clippy::cast_precision_loss,
83                reason = "the magnitude of the numbers we're working on here are too small to lose \
84                anything"
85            )]
86            Transform::from_translation(Vec3::new(
87                0.,
88                PANGRAMS.len() as f32 * 20. + -(i as f32) * 40.,
89                0.,
90            )),
91        ));
92    }
93}
