46/// A component for animating text content.
47///
48/// The `&'static str` field represents the text to animate, while the `usize`
49/// field tracks the current animation state, i.e. how many characters of the
50/// text is currenly being displayed.
51#[derive(Component)]
52struct AnimateText(&'static str, usize);
53
54/// A component for animating the color of text.
55///
56/// The `ColorCurve` field defines the color gradient used for the animation.
57#[derive(Component)]
58struct AnimateColor(ColorCurve<Srgba>);
59
60/// Spawns the text entities for the example and configures gizmo defaults.
61fn setup(
62    mut commands: Commands,
63    assets: Res<DemoAssets>,
64    #[cfg(feature = "gizmos")] store: ResMut<GizmoConfigStore>,
65) {
66    #[cfg(feature = "gizmos")]
67    gizmos::configure_gizmo_defaults(store);
68
69    commands.spawn(Camera2d);
70
71    commands.spawn((
72        AnimateColor(ColorCurve::new(RAINBOW).expect("RAINBOW contains at least two colors")),
73        ImageFontSpriteText::default(),
74        ImageFontText::default()
75            .text(TEXT)
76            .font(assets.example.clone())
77            .font_height(36.0),
78        // Shift 0.5 pixels so our characters end up at integer pixel values. This is only
79        // necessary because we're using a horizontally centered anchor combined with an
80        // odd number of characters. If we used left or right alignment, we wouldn't need
81        // to do this, even with an odd number of characters.
82        Transform::from_translation(Vec3::new(0.5, 0., 0.)),
83    ));
84    commands.spawn((
85        AnimateText(TEXT, 0),
86        // This demonstrates using the `anchor` field on the `ImageFontSpriteText`.
87        // To still center it horizontally with the text above, we shift it left by half its width.
88        ImageFontSpriteText::default()
89            .color(tailwind::AMBER_500)
90            .anchor(Anchor::CenterLeft),
91        ImageFontText::default().font(assets.example.clone()),
92        #[expect(
93            clippy::cast_precision_loss,
94            reason = "the magnitude of the numbers we're working on here are too small to lose \
95            anything"
96        )]
97        Transform::from_translation(Vec3::new(
98            -((TEXT.chars().count() * FONT_WIDTH / 2) as f32),
99            40.,
100            0.,
101        )),
102    ));
103
104    commands.spawn((
105        ImageFontSpriteText::default()
106            .color(tailwind::ZINC_500)
107            .letter_spacing(LetterSpacing::Pixel(2)),
108        ImageFontText::default()
109            .text(TEXT)
110            .font(assets.example.clone())
111            .font_height(36.0),
112        Transform::from_translation(Vec3::new(0.5, -40., 0.)),
113    ));
114
115    // This will currently render without spaces; I intend to add a new feature
116    // shortly that will allow a font to have space without needing to have a
117    // space in the font image itself.
118    commands.spawn((
119        AnimateColor(ColorCurve::new(RAINBOW).expect("RAINBOW contains at least two colors")),
120        ImageFontSpriteText::default().letter_spacing(LetterSpacing::Pixel(1)),
121        ImageFontText::default()
122            .text(TEXT.to_uppercase())
123            .font(assets.variable_width.clone())
124            .font_height(32.0),
125        Transform::from_translation(Vec3::new(0., -120., 0.)),
126    ));
127}

129/// Animates the text content of entities with the `AnimateText` component.
130///
131/// This system modifies the `ImageFontText` component to display an animated
132/// sequence of characters, cycling through the text content over time.
133fn animate_text(
134    mut query: Query<(&mut AnimateText, &mut ImageFontText)>,
135    time: Res<Time>,
136    mut timer: Local<Timer>,
137) {
138    if timer.duration().is_zero() || timer.mode() == TimerMode::Once {
139        timer.set_duration(Duration::from_secs_f32(0.1));
140        timer.set_mode(TimerMode::Repeating);
141    }
142
143    timer.tick(time.delta());
144    if !timer.just_finished() {
145        return;
146    }
147
148    for (mut animated_text, mut image_font_text) in &mut query {
149        let char_count = animated_text.0.chars().count();
150
151        animated_text.1 += 1;
152        if animated_text.1 > char_count * 2 {
153            animated_text.1 = 0;
154        }
155
156        let show_count = if animated_text.1 > char_count - 1 {
157            char_count - (animated_text.1 - char_count)
158        } else {
159            animated_text.1
160        };
161        image_font_text.text = animated_text.0.chars().take(show_count).collect::<String>();
162    }
163}

165/// Animates the color of text entities with the `AnimateColor` component.
166///
167/// This system modifies the `color` field of the `ImageFontSpriteText`
168/// component to cycle through the colors defined in the `RAINBOW` palette.
169#[expect(
170    clippy::cast_sign_loss,
171    clippy::cast_possible_truncation,
172    clippy::cast_precision_loss,
173    reason = "the magnitude of the numbers we're working on here are too small to lose anything"
174)]
175fn animate_color(mut query: Query<(&AnimateColor, &mut ImageFontSpriteText)>, time: Res<Time>) {
176    for (animate_color, mut image_sprite_font_text) in &mut query {
177        let animation_progress = time.elapsed_secs() / RAINBOW.len() as f32;
178        let len = (RAINBOW.len() - 1) as f32;
179        if animation_progress.trunc() as u32 % 2 == 0 {
180            image_sprite_font_text.color = animate_color
181                .0
182                .sample(animation_progress.fract() * len)
183                .expect("fract() is [0,1) and `len` will always be within bounds")
184                .into();
185        } else {
186            image_sprite_font_text.color = animate_color
187                .0
188                .sample(len - (animation_progress.fract() * len))
189                .expect("fract() is [0,1) and `len` will always be within bounds")
190                .into();
191        }
192    }
193}

195/// Gizmos related example code
196#[cfg(feature = "gizmos")]
197mod gizmos {
198
199    use bevy_image_font::atlas_sprites::gizmos::AtlasSpritesGizmoConfigGroup;
200
201    use super::{ButtonInput, GizmoConfigStore, KeyCode, Res, ResMut};
202
203    /// Configures default gizmo rendering settings.
204    ///
205    /// This function initializes the default visibility settings for
206    /// text-related gizmos, enabling anchor points and bounding box
207    /// rendering by default.
208    ///
209    /// # Default Behavior
210    /// - **Text anchor points**: Enabled (`render_text_anchor_point = true`).
211    /// - **Character anchor points**: Enabled (`render_character_anchor_point =
212    ///   true`).
213    /// - **Character bounding boxes**: Enabled (`render_character_box = true`).
214    ///
215    /// This function is called at startup to ensure gizmos are visible by
216    /// default.
217    pub(super) fn configure_gizmo_defaults(mut store: ResMut<'_, GizmoConfigStore>) {
218        let (_, atlas_sprites_config) = store.config_mut::<AtlasSpritesGizmoConfigGroup>();
219        atlas_sprites_config.render_text_anchor_point = true;
220        atlas_sprites_config.render_character_anchor_point = true;
221        atlas_sprites_config.render_character_box = true;
222    }
223
224    /// Handles keyboard input to toggle gizmo rendering settings.
225    ///
226    /// This function listens for key presses and updates the gizmo
227    /// configuration to enable or disable various debug visuals in real
228    /// time.
229    ///
230    /// # Parameters
231    /// - `input`: Reference to [`ButtonInput<KeyCode>`], used to detect key
232    ///   presses.
233    /// - `store`: Mutable reference to the [`GizmoConfigStore`] to update
234    ///   settings.
235    ///
236    /// # Keyboard Shortcuts
237    /// - **`G`**: Toggles all gizmo rendering on/off.
238    /// - **`A`**: Toggles character anchor point gizmos.
239    /// - **`B`**: Toggles character bounding box gizmos.
240    /// - **`T`**: Toggles text anchor point gizmos.
241    pub(super) fn toggle_gizmos(
242        input: Res<ButtonInput<KeyCode>>,
243        mut store: ResMut<GizmoConfigStore>,
244    ) {
245        let (config, atlas_sprites_config) = store.config_mut::<AtlasSpritesGizmoConfigGroup>();
246        if input.just_pressed(KeyCode::KeyG) {
247            config.enabled = !config.enabled;
248            println!("Gizmos enabled: {}", config.enabled);
249        }
250        if input.just_pressed(KeyCode::KeyA) {
251            atlas_sprites_config.render_character_anchor_point =
252                !atlas_sprites_config.render_character_anchor_point;
253            println!(
254                "Character anchor point gizmo enabled: {}",
255                atlas_sprites_config.render_character_anchor_point
256            );
257        }
258        if input.just_pressed(KeyCode::KeyB) {
259            atlas_sprites_config.render_character_box = !atlas_sprites_config.render_character_box;
260            println!(
261                "Character box gizmo enabled: {}",
262                atlas_sprites_config.render_character_box
263            );
264        }
265        if input.just_pressed(KeyCode::KeyT) {
266            atlas_sprites_config.render_text_anchor_point =
267                !atlas_sprites_config.render_text_anchor_point;
268            println!(
269                "Text anchor gizmo enabled: {}",
270                atlas_sprites_config.render_text_anchor_point
271            );
272        }
273    }
