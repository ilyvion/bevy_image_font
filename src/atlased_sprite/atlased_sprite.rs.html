48/// A component for animating text content.
49///
50/// The `&'static str` field represents the text to animate, while the `usize`
51/// field tracks the current animation state, i.e. how many characters of the
52/// text is currenly being displayed.
53#[derive(Component)]
54struct AnimateText(&'static str, usize);
55
56/// A component for animating the color of text.
57///
58/// The `ColorCurve` field defines the color gradient used for the animation.
59#[derive(Component)]
60struct AnimateColor(ColorCurve<Srgba>);

62/// Spawns the text entities for the example and configures gizmo defaults.
63fn setup(
64    mut commands: Commands,
65    assets: Res<DemoAssets>,
66    #[cfg(feature = "gizmos")] store: ResMut<GizmoConfigStore>,
67) {
68    #[cfg(feature = "gizmos")]
69    common::gizmos::configure_gizmo_defaults(store);
70
71    commands.spawn(Camera2d);
72
73    commands.spawn((
74        AnimateColor(ColorCurve::new(RAINBOW).expect("RAINBOW contains at least two colors")),
75        ImageFontSpriteText::default(),
76        ImageFontText::default()
77            .text(TEXT)
78            .font(assets.example.clone())
79            .font_height(36.0),
80        // Shift 0.5 pixels so our characters end up at integer pixel values. This is only
81        // necessary because we're using a horizontally centered anchor combined with an
82        // odd number of characters. If we used left or right alignment, we wouldn't need
83        // to do this, even with an odd number of characters.
84        Transform::from_translation(Vec3::new(0.5, 0., 0.)),
85    ));
86    commands.spawn((
87        AnimateText(TEXT, 0),
88        // This demonstrates using the `anchor` field on the `ImageFontSpriteText`.
89        // To still center it horizontally with the text above, we shift it left by half its width.
90        ImageFontSpriteText::default()
91            .color(tailwind::AMBER_500)
92            .anchor(Anchor::CenterLeft),
93        ImageFontText::default().font(assets.example.clone()),
94        #[expect(
95            clippy::cast_precision_loss,
96            reason = "the magnitude of the numbers we're working on here are too small to lose \
97            anything"
98        )]
99        Transform::from_translation(Vec3::new(
100            -((TEXT.chars().count() * FONT_WIDTH / 2) as f32),
101            40.,
102            0.,
103        )),
104    ));
105
106    commands.spawn((
107        ImageFontSpriteText::default()
108            .color(tailwind::ZINC_500)
109            .letter_spacing(LetterSpacing::Pixel(2)),
110        ImageFontText::default()
111            .text(TEXT)
112            .font(assets.example.clone())
113            .font_height(36.0),
114        Transform::from_translation(Vec3::new(0.5, -40., 0.)),
115    ));
116
117    // This will currently render without spaces; I intend to add a new feature
118    // shortly that will allow a font to have space without needing to have a
119    // space in the font image itself.
120    commands.spawn((
121        AnimateColor(ColorCurve::new(RAINBOW).expect("RAINBOW contains at least two colors")),
122        ImageFontSpriteText::default().letter_spacing(LetterSpacing::Pixel(1)),
123        ImageFontText::default()
124            .text(TEXT.to_uppercase())
125            .font(assets.variable_width.clone())
126            .font_height(32.0),
127        Transform::from_translation(Vec3::new(0., -120., 0.)),
128    ));
129}

131/// Animates the text content of entities with the `AnimateText` component.
132///
133/// This system modifies the `ImageFontText` component to display an animated
134/// sequence of characters, cycling through the text content over time.
135fn animate_text(
136    mut query: Query<(&mut AnimateText, &mut ImageFontText)>,
137    time: Res<Time>,
138    mut timer: Local<Timer>,
139) {
140    if timer.duration().is_zero() || timer.mode() == TimerMode::Once {
141        timer.set_duration(Duration::from_secs_f32(0.1));
142        timer.set_mode(TimerMode::Repeating);
143    }
144
145    timer.tick(time.delta());
146    if !timer.just_finished() {
147        return;
148    }
149
150    for (mut animated_text, mut image_font_text) in &mut query {
151        let char_count = animated_text.0.chars().count();
152
153        animated_text.1 += 1;
154        if animated_text.1 > char_count * 2 {
155            animated_text.1 = 0;
156        }
157
158        let show_count = if animated_text.1 > char_count - 1 {
159            char_count - (animated_text.1 - char_count)
160        } else {
161            animated_text.1
162        };
163        image_font_text.text = animated_text.0.chars().take(show_count).collect::<String>();
164    }
165}

167/// Animates the color of text entities with the `AnimateColor` component.
168///
169/// This system modifies the `color` field of the `ImageFontSpriteText`
170/// component to cycle through the colors defined in the `RAINBOW` palette.
171#[expect(
172    clippy::cast_sign_loss,
173    clippy::cast_possible_truncation,
174    clippy::cast_precision_loss,
175    reason = "the magnitude of the numbers we're working on here are too small to lose anything"
176)]
177fn animate_color(mut query: Query<(&AnimateColor, &mut ImageFontSpriteText)>, time: Res<Time>) {
178    for (animate_color, mut image_sprite_font_text) in &mut query {
179        let animation_progress = time.elapsed_secs() / RAINBOW.len() as f32;
180        let len = (RAINBOW.len() - 1) as f32;
181        if (animation_progress.trunc() as u32).is_multiple_of(2) {
182            image_sprite_font_text.color = animate_color
183                .0
184                .sample(animation_progress.fract() * len)
185                .expect("fract() is [0,1) and `len` will always be within bounds")
186                .into();
187        } else {
188            image_sprite_font_text.color = animate_color
189                .0
190                .sample(len - (animation_progress.fract() * len))
191                .expect("fract() is [0,1) and `len` will always be within bounds")
192                .into();
193        }
194    }
195}
