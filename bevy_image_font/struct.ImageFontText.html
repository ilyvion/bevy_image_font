#[non_exhaustive]
pub struct ImageFontText {
    pub text: String,
    pub font: Handle<ImageFont>,
    pub font_height: Option<f32>,
}
Expand description

Text rendered using an ImageFont.

Fields (Non-exhaustive)§

This struct is marked as non-exhaustive
Non-exhaustive structs could have additional fields added in future. Therefore, non-exhaustive structs cannot be constructed in external crates using the traditional Struct { .. } syntax; cannot be matched against without a wildcard ..; and struct update syntax will not work.
§text: String

The string of text to be rendered. Each character in the string is mapped to a corresponding glyph in the associated ImageFont.

§font: Handle<ImageFont>

The handle to the ImageFont used to render this text. The font’s texture and atlas mapping determine how characters are displayed.

§font_height: Option<f32>

If set, overrides the height the font is rendered at. This should be an integer multiple of the ‘native’ height if you want pixel accuracy, but we allow float values for things like animations.

38fn spawn_text(mut commands: Commands, assets: Res<DemoAssets>) {
39    commands.spawn(Camera2d);
40
41    commands.spawn((
42        ImageFontPreRenderedText::default(),
43        ImageFontText::default()
44            .text(TEXT)
45            .font(assets.example.clone())
46            .font_height(36.0),
47        // Our font is 45 pixels wide per character, and with an odd number of characters, the
48        // text aligns to the middle of a pixel, causing imperfect rendering. Shifting the
49        // position by 0.5 pixels ensures alignment to pixel boundaries, resulting in crisp,
50        // pixel-perfect rendering.
51        Transform::from_translation(Vec3::new(0.5, 0., 0.)),
52    ));
53    commands.spawn((
54        ImageFontPreRenderedText::default(),
55        ImageFontText::default()
56            .text(TEXT)
57            .font(assets.example.clone()),
58        // Instead of shifting the character by 0.5 pixels when the text lands in the middle of
59        // a pixel, we can anchor the sprite to an edge and move it by a whole number of pixels.
60        // To center it with the text above, we shift it left by half its width.
61        Sprite {
62            anchor: Anchor::CenterLeft,
63            ..default()
64        },
65        #[expect(
66            clippy::cast_precision_loss,
67            reason = "the magnitude of the numbers we're working on here are too small to lose \
68            anything"
69        )]
70        Transform::from_translation(Vec3::new(
71            -((TEXT.chars().count() * FONT_WIDTH / 2) as f32),
72            40.,
73            0.,
74        )),
75    ));
76}
